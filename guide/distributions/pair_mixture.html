Generative model¶

n_cells = 500
cell_ids = [str(i) for i in range(n_cells)]

cells = la.Dim(pd.Series(cell_ids, name="cell"))

x_value = pd.Series(np.random.uniform(0, 5, n_cells), index=cells.index)
x = la.Fixed(x_value, label="x")

distributions_gs = {
    "a": la.distributions.Normal(5.0, 0.5),
    "b": la.distributions.Laplace(-1.0, 0.5),
    "c": la.distributions.Normal(10.0, 0.5),
}
distributions_dim = la.Dim(list(distributions_gs.keys()), "distribution")

w = pd.Series(
    [0.1, 0.5, 0.4],
    index=distributions_dim.index
)
w = la.Fixed(w)

mixture = la.distributions.Mixture(distributions = distributions_gs, weight=w, label="mixture", definition = la.Definition([cells]))

mixture.plot()

Test whether the probability of a component is the same as the mixture with weight = 1

x = torch.tensor(5.)
for distribution_id in distributions_dim.coords:
    w = pd.Series(0., index = distributions_dim.index)
    w.loc[distribution_id] = 1.
    w = la.Fixed(w)
    
    mixture = la.distributions.Mixture(distributions = distributions_gs, weight=w, label="mixture")
    mixture.run_recursive()
    
    assert mixture.log_prob(x) == getattr(mixture, distribution_id).log_prob(x), distribution_id

Pair mixture¶

distributions_gs = {
    "a": la.distributions.Normal(5.0, 0.5),
    "b": la.distributions.Laplace(-1.0, 0.5)
}

w = la.Fixed(0.)

mixture = la.distributions.PairMixture(
    distribution1 = distributions_gs["b"],
    distribution0 = distributions_gs["a"],
    weight=w,
    label="mixture"
)
mixture.run_recursive()

assert mixture.log_prob(x) == mixture.distribution0.log_prob(x)

w = la.Fixed(1.)

mixture = la.distributions.PairMixture(
    distribution1 = distributions_gs["b"],
    distribution0 = distributions_gs["a"],
    weight=w,
    label="mixture"
)
mixture.run_recursive()

assert mixture.log_prob(x) == mixture.distribution1.log_prob(x)

w = la.Fixed(0.5)

mixture = la.distributions.PairMixture(
    distribution1 = distributions_gs["b"],
    distribution0 = distributions_gs["a"],
    weight=w,
    label="mixture"
)
mixture.run_recursive()

assert torch.isclose(
    mixture.log_prob(x),
    torch.log(torch.exp(mixture.distribution1.log_prob(x)) * 0.5 + torch.exp(mixture.distribution0.log_prob(x)) * 0.5)
)
