import latenta as la
import numpy as np
import pandas as pd

observed_value = pd.Series(np.random.normal(2., 1, (100, )), name = "observed")
observed_value.index.name = "sample"

observed_definition = la.Definition.from_dataarray(observed_value)

loc_prior = la.distributions.Normal()
loc = la.Latent(loc_prior)

scale_prior = la.distributions.LogNormal()
scale = la.Latent(scale_prior)

dist = la.distributions.Normal(loc = loc, scale = scale)

observed = la.Observed(observed_value, dist)

la.interpretation.ComponentGraph(observed)

<latenta.interpretation.viz.ComponentGraph at 0x1c33dcb20d0>

inference = la.infer.svi.SVI(observed, [la.infer.loss.ELBO()], la.infer.optim.Adam(lr = 0.01))
trainer = la.infer.trainer.Trainer(inference)

trace = trainer.train(10000)
trace.plot();
