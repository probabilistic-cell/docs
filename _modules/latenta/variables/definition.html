def is_definition(value):
    return isinstance2(value, Definition)



[docs]class Definition(LabelledMixin):
    """
    Defines how a variable should or will look like
    """

    id = None

    def __init__(
        self,
        dims=None,
        label: str = None,
        symbol: str = None,
        description: str = None,
        dependencies: dict = None,
    ):
        if (label is None) and (is_definition(dims)):
            label = dims.labelling._label
        if (symbol is None) and (is_definition(dims)):
            symbol = dims.labelling._symbol
        if (description is None) and (is_definition(dims)):
            description = dims.labelling._description
        if is_definition(dims):
            dependencies = dims.labelling.dependencies

        super().__init__(label, symbol, description, dependencies=dependencies)

        if dims is None:
            dims = []
        if is_dim(dims):
            dims = [dims]

        # check dims
        for dim in dims:
            assert is_dim(dim), dim

        self._dims = list(dims)

        self._index = {dim.id: i for i, dim in enumerate(self._dims)}

    def __iter__(self):
        return self._dims.__iter__()

    def __len__(self):
        return self._dims.__len__()

    @property
    def ndim(self):
        return len(self._dims)

    @property
    def dims(self):
        return tuple(dim.id for dim in self._dims)

    @property
    def shape(self):
        return tuple(dim.size for dim in self._dims)

    @property
    def coords(self):
        return {dim.id: dim.coords for dim in self._dims if dim.coords is not None}

    def index(self, dim):
        return self._index[dim.id]

    def expand(self, *dims):
        assert (is_dim(dim) for dim in dims)
        return Definition([*dims, *self._dims])

    def expand_right(self, *dims):
        assert (is_dim(dim) for dim in dims)
        return Definition([*self._dims, *dims])

    def replace_dim(self, from_dim, to_dim):
        assert is_dim(from_dim) and is_dim(to_dim)
        assert from_dim in self._dims
        return Definition([dim if dim != from_dim else to_dim for dim in self._dims])

    def clone(self):
        return Definition(self, label=self._label, symbol=self._symbol)

    def tensor_to_xr(self, value):
        value_definition = self.value_definition
        return xr.DataArray(
            value.detach().cpu().numpy(),
            dims=value_definition.dims,
            coords={dim: list(coord) for dim, coord in value_definition.coords.items()},
            name=value_definition.label,
        )

    def __getitem__(self, key):
        if isinstance(key, int):
            return self._dims[key]
        elif isinstance(key, str):
            return self._dims[self.dims.index(key)]
        return Definition(self._dims[key])

    def __setitem__(self, key, value):
        if isinstance(key, int):
            self._dims[key] = value
        elif isinstance(key, str):
            self._dims[self.dims.index(key)] = value

    def to_xr(self, value, broadcast=False):
        if isinstance(value, pd.Series):
            assert (
                value.index.name is not None
            ), "When providing a series, the index has to have a name"
            value = xr.DataArray.from_series(value)
        elif isinstance(value, pd.DataFrame):
            assert (
                value.index.name is not None
            ), "When providing a dataframe, the index has to have a name"
            assert (
                value.columns.name is not None
            ), "When providing a dataframe, the columns has to have a name"
            value = xr.DataArray(value)
        elif isinstance(value, (float, int)) or np.isscalar(value):
            value = xr.DataArray(value)
        elif isinstance(value, np.ndarray):
            given_dims = self.dims[: value.ndim]
            value = xr.DataArray(
                value,
                dims=given_dims,
                coords={
                    dim: list(self.coords[dim])
                    for dim in given_dims
                    if dim in self.coords
                },
            )
        elif torch.is_tensor(value):
            assert self.dims is not None
            assert self.coords is not None
            given_dims = self.dims[: len(value.shape)]
            value = xr.DataArray(
                value.detach().cpu().numpy().copy(),
                dims=given_dims,
                coords={
                    dim: list(self.coords[dim])
                    for dim in given_dims
                    if dim in self.coords
                },
            )
        elif isinstance(value, xr.DataArray):
            pass
        else:
            raise ValueError(
                f"\nUnsupported value type: {type(value)}.\nValue can be a scalar, numpy array, pandas Series/DataFrame, a torch.tensor, or an xarray.DataArray"
            )
        assert isinstance(value, xr.DataArray)

        if value.dtype == "float64":
            value = value.astype(np.float32)
        if value.dtype == "int64":
            value = value.astype(np.int32)

        if broadcast:
            value = value.broadcast_like(self.get_reference())
            assert value.shape == self.shape

        assert all([dim in self.dims for dim in value.dims]), (
            self,
            self.dims,
            value.dims,
        )
        return value

    @classmethod
    def from_xr(cls, value, **kwargs):
        assert not isinstance(value, np.ndarray), "Dimensions should be provided"

        if torch.is_tensor(value):
            value = value.item()
        if isinstance(value, (int, float)) or np.isscalar(value):
            value = xr.DataArray(value)
        elif isinstance(value, pd.Series):
            assert value.index.name is not None
            value = xr.DataArray.from_series(value)
        elif isinstance(value, pd.DataFrame):
            assert value.index.name is not None
            assert value.columns.name is not None
            value = xr.DataArray(value)
        elif isinstance(value, xr.DataArray):
            pass
        else:
            raise ValueError(f"Cannot infer definition from {type(value)}")

        dims = []
        for dim, shape in zip(value.dims, value.shape):
            if dim in value.coords:
                coords = list(value.coords[dim].values)
                dim = Dim(pd.DataFrame(index=pd.Series(coords, name=dim)))
            else:
                dim = Dim(pd.DataFrame(index=pd.Series(range(shape), name=dim)))
            dims.append(dim)

        return cls(dims, **{**{"label": value.name}, **kwargs})

    def get_reference(self):
        return xr.DataArray(
            np.zeros(self.shape),
            dims=self.dims,
            coords={dim: list(coord) for dim, coord in self.coords.items()},
        )

    def subset(self, *args):
        assert all(
            arg in self for arg in args
        ), f"Some dimensions are not found: {args} vs {self.dims}"
        return Definition([dim for dim in self if dim in args])

    def contract(self, *args):
        assert all(
            arg in self for arg in args
        ), f"Some dimensions are not found: {args} vs {self.dims}"
        return Definition([dim for dim in self if dim not in args])

    def filter(self, **kwargs):
        for dim, coords in kwargs.items():
            assert dim in self.dims

            if not is_dim_broadcasted(self._dims[self.dims.index(dim)]):
                self._dims[self.dims.index(dim)] = Dim(coords, dim)
        return self

    def broadcaster_to(self, old_definition):
        return TensorMapper(old_definition, self)

    def indexer(self):
        return TensorIndexer(self.dims)


[docs]    def broadcast(self, *args):
        """
        Makes the size of certain dimensions None, ready to be broadcasted
        """
        assert all(
            arg in self for arg in args
        ), f"Some dimensions are not found: {args} vs {self.dims}"
        if args is None:
            args = self._dims
        self._dims = [dim.broadcast() if dim in args else dim for dim in self._dims]
        return self


    def autobroadcast(self, definitions):
        dims = {}
        for definition in definitions:
            dims.update({dim.id: dim for dim in definition if dim in self})
        return self.broadcast(*dims.values())

    def __repr__(self):
        return self.__str__()

    def __str__(self):
        return self.repr_unicode()

    def _repr_shape_unicode_(self):
        return (
            "["
            + ", ".join([size if size is not None else "•" for size in self.shape])
            + "]"
        )

    def repr_dims_unicode(self):
        template = "{dim} ({size})"
        return (
            "["
            + ", ".join(
                [
                    template.format(dim=dim, size=size if size is not None else "•")
                    for dim, size in zip(self.dims, self.shape)
                ]
            )
            + "]"
        )

    def _repr_html_(self):
        return f"<strong>{self.label}</strong>: Definition {self.repr_dims_unicode()}"

    def repr_unicode(self):
        return f"{self.label}: Definition {self.repr_dims_unicode()}"

    def define(self) -> Definition:
        return Definition(
            self._dims,
            label=self._label,
            symbol=self._symbol,
            description=self._description,
        )

    def redimension(self, dims) -> Definition:
        return Definition(
            dims, label=self._label, symbol=self._symbol, description=self._description
        )

    @property
    def clean(self):
        definition = Definition(self)
        definition.label = None
        definition.symbol = None
        return definition

    def node_attr(self, theme):
        return theme["node"]["default"]

    def node_label(self, theme):
        label = ""
        from latenta.introspect import (
            latex_to_node_label,
        )  # pylint: disable=import-outside-toplevel

        label += latex_to_node_label(self._repr_latex_short_())

        # support
        if ("support" in theme["node"]["default"]) and (
            theme["node"]["default"]["support"]
        ):
            support = True

        support = False
        if support:
            label += " ∈ " + self.support.repr_unicode()

        # dims
        dims = False
        if dims:
            label += " " + self.repr_dims_unicode()
        return label

    def node_legend(self):
        return {"type": "default", "label": ""}

    def edge_attr(self, component, theme):
        return {
            attr: value
            for attr, value in theme["edge"]["default"].items()
            if attr not in ["label"]
        }

    def __hash__(self):
        return hash(self.uuid)


[docs]    def replace_dims(self, **kwargs):
        """
        Replace one or more dimensions
        """
        for dim_id, dim in kwargs.items():
            assert dim_id in self.dims
            self._dims[self.dims.index(dim_id)] = dim
        return self


    def inv_transform(self, transforms):
        definition = self
        for transform in transforms[::-1]:
            definition = transform.invert_definition(definition)
        return definition

    @classmethod
    def from_definitions(cls, definitions):
        dims = {}
        for definition in definitions:
            dims.update({dim.id: dim for dim in definition})
        return cls(list(dims.values()))
