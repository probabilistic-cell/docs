[docs]class Sine(Linear):
    shift = CoefficientPromise(
        "shift", r"\zeta", required=False, default_distribution="x", default=None
    )

    def __call__(self, x, a=1, b=0.0, shift=0.0):
        out = b + a * (torch.sin(x - shift) + 1) / 2
        assert not torch.any(torch.isnan(out)), f"NaNs in output of {self.label}"
        return out

    def _repr_formula_latex_(self):
        return self._repr_formula_latex_short_()

    def _repr_formula_latex_short_(self):
        return self._format_latex(r"[a] \cdot \sin([x] - [shift])")




[docs]class Cartesian(Scalar):
    """
    Converting polar to cartesian coordinates
    """

    x = VariablePromise("x", r"angle", dims=ParentContractedRight())
    radius = VariablePromise(
        "radius", "r", required=False, dims=ParentContractedRight()
    )

    _define_expand_right = {"axis": Dim(["x", "y"], "axis")}

    invertible = True
    bijective = False

    def __call__(self, x, radius=1.0):
        return torch.stack([torch.cos(x) * radius, torch.sin(x) * radius], -1)

    def _repr_formula_latex_(self):
        return self._format_latex(r"""\left[ cos([x])\;\;sin([x]) \right]""")

    _repr_formula_latex_short_ = _repr_formula_latex_

    def invert(self, y):
        out = torch.atan2(y[..., 1], y[..., 0]) % (math.tau)
        assert not torch.any(torch.isnan(out)), f"NaNs in output of {self.label}"
        return out

    def log_abs_det_jacobian(self, x, y):
        # the determinant of the jacobian of polar to cartesian is r(cos^2 x - sin^2 x) => log abs det = log(r)
        # the determinant of the reverse transformation is thus -log(r)

        return torch.log(torch.sqrt((x ** 2).sum(-1))[..., None])

    def log_prob(self, y):
        raise NotImplementedError()
        x = self.invert(y)
        return self.x.log_prob(x) - self.log_abs_det_jacobian(x, y)

    @property
    def likelihood(self):
        return self.x.likelihood() - self.log_abs_det_jacobian(self.x.value, self.value)




[docs]class CircularShift(Scalar):
    sign = VariablePromise("sign", r"\textit{sign}")
    shift = VariablePromise("shift", r"\textit{shift}")

    def __call__(self, x, shift=0.0, sign=1.0):
        return ((x + shift) * sign) % (math.tau)

    def invert(self, y, shift=0.0, sign=1.0):
        return (y / sign) - shift

    def _log_prob(self, x, y, log_prob, shift=0.0, sign=1.0):
        return log_prob
