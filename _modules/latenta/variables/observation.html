import numpy as np

def is_observation(value):
    return isinstance2(value, Observation)


[docs]class Observation(RandomMixin, Fixed):
    """
    An observed random variable, i.e. the value of this variable is uncertain (=follows a distribution $p$) but observed (=we have a sample from this distribution). The parameters of the variables' distribution $p$ may depend on other observed, latent, fixed or parameter variables. using `likelihood`
    """

    has_scale = True

    _define_excluded = set(["p", "weights"])

    p = RandomPromise("p", "p", "The distribution followed by this observation.")
    weights = VariablePromise("weights", "w", "The weights associated with each observation.", dims = ParentBroadcasted())

    def __init__(self, value, p, *args, **kwargs):
        super().__init__(value = value, p = p, *args, **kwargs)


[docs]    def run(self):
        values = {}
        for component_id, component in self.components.items():
            if component_id == "p":
                values[component_id] = None
            else:
                values[component_id] = component.value # We cannot broadcast here

        value = self.__call__(**values)

        for transform in self.transforms:
            value = transform.run(value)

        self.value = value


    def __call__(self, p, weights = 1.):
        return super().__call__()

    @property
    def likelihood(self):
        """
        The likelihood of the observation according to $p$
        """
        log_prob = self.p.log_prob(self.value)

        if "weights" in self.component_ids:
            w = self._broadcasters["weights"](self.weights.value)

            return log_prob * w

        return log_prob

    @property
    def likelihood_sum(self):
        """
        The marginal likelihood according to $p$
        """
        # we have to scale this because of possible subsampling
        return self.likelihood.sum() * (np.prod(self.shape) / self.value.numel())

    def node_attr(self, theme):
        return {**super().node_attr(theme), **theme["node"]["observation"]}

    def node_legend(self):
        return {"type":"observation", "label":"Observation"}

    @property
    def event_dims(self):
        """
        The event dimensions of the observation
        """
        return self.p.event_dims

    def get_scale(self):
        return self.value.numel()
