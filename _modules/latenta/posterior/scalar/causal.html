def plot_scalar_scalar(ax, x, y, legend_artists={}, q=(0.99, 0.95, 0.9, 0.8, 0.65)):
    q_left = [0.99, 0.95, 0.9, 0.8, 0.65]
    q = q_left + [0.5] + [round(1 - q, 3) for q in q_left[::-1]]
    qy = np.quantile(y, q, 0)
    qmed = qy[len(q_left)]

    y_range = (y.min(), y.max())
    y_scale = y.max() - y.min()

    n = len(q_left)

    for i, color in enumerate(
        mpl.colors.LinearSegmentedColormap.from_list(
            "green2red", ["#5898c5ff", "white"]
        )(np.linspace(0, 1, n + 1)[:-1][::-1])
        # sns.color_palette("PuBu", n + 1)[:-1]
        # ['#e34a3322','#e34a3322','#e34a3322','#e34a3322','#e34a3322','#e34a3322'][: len(q_left)]
        # [lighten_color("#5898c5ff", 1.5 * (i) / (n + 2)) for i in range(n)]
        # ['#fef0d955','#fdd49e55','#fdbb8455','#fc8d5955','#e34a3355','#b3000055'][: len(q_left)]
        # ["#0059a822", "#0059a822", "#0059a822", "#0059a822", "#0059a822"][: len(q_left)]
        # ["#00000022", "#00000022", "#00000022", "#00000022", "#00000022"][: len(q_left)]
        # ["#DDDDDD", "#CCCCCC", "#AAAAAA", "#888888", "#666666"][: len(q_left)]
    ):
        artist = ax.fill_between(x, qy[i], qy[-i - 1], color=color, lw=0, alpha=1.0)
        legend_artists[f"[{q[i]:.0%}, {q[-i-1]:.0%}]"] = artist
    artist = ax.plot(x, qmed, color="#084081", lw=3)
    legend_artists["median"] = artist[0]


def plot_scalar_scalar_empirical(ax, x, y, legend_artists, x_scale, xmin=0):
    sns.scatterplot(
        x=x.to_pandas(), y=y.to_pandas(), ax=ax, s=10, color="#999999", clip_on=False
    )
    zero = x == xmin
    if zero.sum() > 10:
        sns.violinplot(
            x=x.loc[zero].values,
            y=y.loc[zero].values,
            ax=ax,
            width=x_scale / 10,
            lw=0,
            cut=0,
        )


def plot_scalar_scalar_gs(ax, x, y):
    sns.lineplot(x=x, y=y, ax=ax, color="red")

[docs]class ScalarScalarCausal(Causal):
    """
    Comparing a scalar input to a scalar output
    """

    def plot(
        self, interpretable=None, ax=None, fig=None, show_empirical=None, show=False
    ):
        x = self.samples[self.input].mean("sample").to_pandas().values
        x_range = (x.min(), x.max())
        x_scale = x.max() - x.min()

        if interpretable is None:
            interpretable = self._get_default_interpretable()
        y = self.samples[interpretable].to_pandas()

        if ax is None:
            fig, (ax,) = plotting.axes_wrap(1)

        legend_artists = {}

        # actual values
        plot_scalar_scalar(ax, x, y, legend_artists=legend_artists)

        # empirical
        if show_empirical is None:
            show_empirical = (self.empirical is not None) and (
                interpretable in self.empirical.samples
            )

        if show_empirical:
            assert self.empirical is not None, "First do an empirical sampling"
            plot_scalar_scalar_empirical(
                ax,
                self.empirical.samples[self.input].sel(sample=0),
                self.empirical.samples[interpretable].sel(sample=0),
                x_scale=x_scale,
                xmin=x_range[0],
                legend_artists=legend_artists,
            )

        # annotate x axis
        annotate_scalar_input_axis(self.input, ax, "x", x_range)

        # annotate y axis
        annotate_scalar_output_axis(
            interpretable, ax, "y", [y.mean(0).values.min(), y.mean(0).values.max()]
        )

        ax.set_title(interpretable.label)

        if fig is not None:
            fig.legend(
                legend_artists.values(), legend_artists.keys(), loc=(0.0, 0), ncol=6
            )

        if not show:
            plt.close(fig)

        return fig




[docs]class ScalarVectorCausal(XVectorCausal):
    """
    Comparing a scalar input to a vector output
    """

    def plot_features(
        self,
        interpretable=None,
        feature_ids=None,
        n_features=10,
        axes=None,
        show_empirical=None,
        legend=True,
        show=False,
    ):
        if interpretable is None:
            interpretable = self._get_default_interpretable()

        x = self.samples[self.input].mean("sample").to_pandas().values
        x_range = (x.min(), x.max())
        x_scale = x.max() - x.min()
        Y = self.samples[interpretable]

        assert interpretable.ndim == 2, interpretable

        feature_dim = interpretable.value_definition[-1]

        if feature_dim.name not in self.samples[interpretable].dims:
            self.samples[interpretable] = self.samples[interpretable].expand_dims(
                {feature_dim.name: [1]}
            )
            feature_ids = [1]
        else:
            assert feature_dim.name in Y.coords

        # determine which features to plot
        # when likelihood_ratios were calculated (e.g. through .sample_bootstrap()) we use the top features
        # otherwise take the first n features
        if feature_ids is None:
            if self.default_perturbed is not None:
                feature_ids = (
                    self.likelihood_ratio.to_pandas()
                    .sort_values(ascending=False)
                    .head(n_features)
                    .index.tolist()
                )
            else:
                feature_ids = (
                    self.samples[interpretable]
                    .coords[feature_dim.name]
                    .values[:n_features]
                )

        # create axes if not provided
        if axes is None:
            fig, axes, legend_ax = plotting.axes_wrap(len(feature_ids), legend=True)
        else:
            fig = axes[0].figure

        # show empirical
        if show_empirical is None:
            show_empirical = (self.empirical is not None) and (
                interpretable in self.empirical.samples
            )

        # plot individual features
        for feature_id, ax in zip(feature_ids, axes):
            legend_artists = {}

            y = Y.sel({feature_dim.name: feature_id}).to_pandas()

            plot_scalar_scalar(ax, x, y, legend_artists=legend_artists)

            if show_empirical:
                assert self.empirical is not None, "First do an empirical sampling"
                # assert interpretable.has_empirical

                assert interpretable in self.empirical.samples
                # assert self.empirical.samples[interpretable].attrs["empirical"]
                plot_scalar_scalar_empirical(
                    ax,
                    self.empirical.samples[self.input].sel(sample=0),
                    self.empirical.samples[interpretable].sel(
                        {feature_dim.name: feature_id, "sample": 0}
                    ),
                    x_scale=x_scale,
                    xmin=x_range[0],
                    legend_artists=legend_artists,
                )

            # annotate x axis
            annotate_scalar_input_axis(self.input, ax, "x", x_range)

            # annotate y axis
            annotate_scalar_output_axis(
                interpretable, ax, "y", [y.mean(0).values.min(), y.mean(0).values.max()]
            )

            ax.set_title(feature_dim.label_features(feature_id))

        if legend:
            legend_ax.legend(
                legend_artists.values(), legend_artists.keys(), loc=(0.0, 1), ncol=3
            )

        if not show:
            plt.close(fig)

        return fig
