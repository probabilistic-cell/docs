def get_initial_quantiles_sampling(p, initial_quantile_mapping, n_samples=3):
    from latenta.variables import is_dim_broadcasted
    import math

    # use p's shape
    shape = p.value_definition.shape

    # for accuracy and speed, we want to limit the number of samples
    # so we share the sampling within the batch dim if the dim is not broadcasted in the prior
    initial_shared_ix = [i for i in range(p.ndim) if not is_dim_broadcasted(p[i])]
    initial_nonshared_ix = [i for i in range(len(shape)) if i not in initial_shared_ix]

    # sample
    samples = torch.stack([p.prior() for _ in range(n_samples)])

    # put all samples + shared_ix in first dimension (to get quantiles)
    samples_reshaped = samples.reshape(
        [
            math.prod([n_samples, *[shape[i] for i in initial_shared_ix]]),
            *[shape[i] for i in initial_nonshared_ix],
        ]
    )

    # get initial quantiles (and reshape to get back to the original shape)
    initial_quantiles = []
    for q1, q2 in initial_quantile_mapping:
        initial_quantiles.append(
            [
                torch.tensor(q1),
                samples_reshaped.quantile(q2, 0)
                .reshape(
                    [
                        1 if i in initial_shared_ix else shape[i]
                        for i in range(len(shape))
                    ]
                )
                .expand(shape),
            ]
        )

    return initial_quantiles


def get_initial_quantiles_icdf(p, initial_quantile_mapping):
    initial_quantiles = []
    for q1, q2 in initial_quantile_mapping:
        initial_quantiles.append(
            [
                torch.tensor(q1),
                p.icdf(torch.tensor(q2)),  # .expand(p.value_definition.shape)
            ]
        )
    return initial_quantiles


def invert_quantiles_normal(initial_quantiles, transforms, loc):
    """
    Invert quantiles to normal distribution
    """
    variances = []
    for quantile, y in initial_quantiles:
        if not torch.is_tensor(quantile):
            quantile = torch.tensor(quantile)
        for transform in transforms[::-1]:
            y = transform.prior_invert_transform(y)
        variances.append(((y - loc) / torch.erfinv(2 * quantile - 1)) ** 2 / 2)
    scale = torch.sqrt(torch.stack(variances, -1).mean(-1))
    return scale



[docs]class Latent(VariationalMixin, RandomMixin, Variable):
    """
    A latent random variable i.e. the actual value of this variable is uncertain and unobserved. The variable follows a prior distribution $p$, which can have parameters that may be fixed and/or depend on other variables. The posterior distribution of a latent is approximated using a variational distribution $q$. $q$ is often but not necessarily a Normal distribution, with some additional transformations so that its support matches that of $p$. $q$ has variational parameters that need to be estimated, which are by default done directly through parameter variables, but which can also be estimated from the data using an amortization function.
    """

    q = RandomPromise("q", "q")
    p = RandomPromise("p", "p")

    def __init__(
        self, p, q=None, definition=None, initial=None, initial_quantiles=None, **kwargs
    ):
        assert p is not None
        # assert p can calculate likelihood (i.e. a distribution or something that can invert towards a distribution)

        if definition is None:
            if q is None:
                # take definition from p
                definition = p.value_definition.clean.unlabel()
            else:
                # take definition from q
                definition = q.value_definition.clean.unlabel()
        else:
            # try to coerce
            if not is_definition(definition):
                definition = Definition(definition)
            if q is None:
                assert all(
                    [size is not None for size in definition.shape]
                ), "The definition of a latent variable cannot have broadcasted dimensions"

        p = p.redefine(definition)

        # create q
        if q is None:
            from latenta import distributions

            # get initial from p's mean
            if initial is None:
                if p.has_mean:
                    loc_value = p.prior_mean
                elif p.has_median:
                    loc_value = p.median
                else:
                    raise ValueError(
                        "Cannot initialize from a distribution without a mean or median, provide initial manually"
                    )
            else:
                loc_value = torch.tensor(
                    definition.to_xr(initial, broadcast=True).values
                )

            assert torch.all(~torch.isnan(loc_value)), "Initial value contains NaNs"

            assert loc_value.shape == definition.shape, (
                loc_value.shape,
                definition.shape,
            )

            # get transforms to go from the prior's support to real
            transforms = p.biject_to()

            # invert the transforms to get the actual loc and definition
            unconstrained_definition = definition.clean.unlabel()
            for transform in transforms[::-1]:
                # invert the definition
                inverted_definition = transform.invert_definition(
                    unconstrained_definition.clone()
                )

                # initialize the transformation
                transform.initialize(inverted_definition, unconstrained_definition)

                # invert the initial
                loc_value = transform.prior_invert_transform(loc_value)
                assert torch.all(torch.isfinite(loc_value)), (loc_value, transform)

                unconstrained_definition = inverted_definition

            loc = Parameter(loc_value, unconstrained_definition)

            # get the scale for the variational distribution:
            # - if initial_quantiles is given use that
            # - if initial_quantiles is not given, can only generate initial_quantiles if initial is not given
            #   - if distribution has icdf, we can directly get quantiles
            #   - if distribution has no icdf, we take samples and then get an icdf from that
            # - if intial_quantiles is available: invert the quantiles towards the normal distribution
            # - if no initial_quantiles available: simply choose scale = 0.1 🤷‍♂️.
            #     this is often not a horrible choice, given that a lot of distributions include scaling factors in their biject_to transforms
            initial_quantile_mapping = [[0.9, 0.65], [0.1, 0.35]]
            if initial_quantiles is None:
                if initial is None:
                    if p.has_icdf:
                        initial_quantiles = get_initial_quantiles_icdf(
                            p, initial_quantile_mapping=initial_quantile_mapping
                        )
                    else:
                        initial_quantiles = get_initial_quantiles_sampling(
                            p, initial_quantile_mapping=initial_quantile_mapping
                        )

            if initial_quantiles is not None:
                assert len(initial_quantiles) > 0
                scale_value = invert_quantiles_normal(
                    initial_quantiles, transforms=transforms, loc=loc_value
                )
            else:
                scale_value = 0.1

            scale = Parameter(
                scale_value,
                unconstrained_definition,
                transforms=distributions.Exponential().biject_to(),
            )

            # create q
            q = distributions.Normal(
                loc,
                scale,
                definition=definition.clean,
                transforms=transforms,
                dependent_dims=p.event_dims,
            )

        if self.has_posterior:
            assert is_distribution(q), "q needs to be a distribution"

        self.value_definition = definition

        super().__init__(definition=definition.clone(), p=p, q=q, **kwargs)

    def __call__(self, p, q):
        out = p
        return out

    def run_local(self):
        """
        :meta private:
        """
        self.value = self.q.value


[docs]    def run_conditional(self, value=None, conditional_dimensions=None, condition=None):
        """
        Conditional latent variable: will take mean of the prior if necessary
        """
        value_set = False
        if conditional_dimensions is not None:
            if any(dim_id in self.dims for dim_id in conditional_dimensions.keys()):

                desired_value_definition = self.value_definition.replace_dims(
                    **{
                        dim_id: dim
                        for dim_id, dim in conditional_dimensions.items()
                        if dim_id in self.dims
                    }
                )

                #! TODO: check actual coords for equality!!
                if desired_value_definition.shape != self.value_definition.shape:
                    value_set = True
                    if condition is not None:
                        raise NotImplementedError("# TODO 😶")

                    # take the prior, broadcast to the latent's dimensions and expand to match the shape
                    self.value = self._broadcasters["p"](self.p.prior_mean)
                    shape = [
                        size
                        if dim_id not in conditional_dimensions
                        else conditional_dimensions[dim_id].size
                        for size, dim_id in zip(self.value_definition.shape, self.dims)
                    ]
                    self.value = self.value.expand(shape)
                    self.value_definition = desired_value_definition
        if not value_set:
            super().run_conditional(
                value=value,
                conditional_dimensions=conditional_dimensions,
                condition=condition,
            )


    @property
    def posterior(self):
        return self.q.likelihood

    @property
    def likelihood(self):
        return super().likelihood

    def _get_value_definition(self):
        assert self.q.value_definition is not None
        return self.q.value_definition.clone()

    def to_variational(self, distribution=None):
        self.q.reset()
        q = self.q.clone()
        q.run()
        q.loc = q.loc.to_fixed()
        q.scale = q.scale.to_fixed()

        if distribution is None:
            distribution = self.p

        fixedlatent = Variational(q, definition=self, distribution=distribution)
        fixedlatent.points = self.points
        fixedlatent.intervals = self.intervals

        return fixedlatent

[docs]class Variational(VariationalMixin, Variable):
    """
    A latent variable with only a variational distribution and no prior. This is often only used when the latent variable has been inferred on another dataset, and we want to keep the variable's uncertainty but don't need its prior anymore.
    """

    q = VariablePromise("q")

    def __init__(self, q, definition, **kwargs):
        super().__init__(definition=definition, **kwargs)
        self.q = q

    def __call__(self, q):
        return q

    def _get_value_definition(self):
        assert self.q.value_definition is not None
        return self.q.value_definition.clone()



class LatentObservation(Latent):
    """
    An observed random variable, i.e. the value of this variable is uncertain (=follows a distribution $p$) but we can sample from it (=$q$). Typically used when trying to simplify a complex model $q$ using a simpler representation $p$.
    """

    q = VariablePromise("q", "q")
    p = RandomPromise("p", "p")

    has_posterior = False

    def node_attr(self, theme):
        return {**super().node_attr(theme), **theme["node"]["latent_observation"]}
