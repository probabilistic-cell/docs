def isobserved_posterior(instance):
    return ".Observed" in repr(instance.__class__.__mro__)


class Conditional(Posterior):
    interpretable = None

    def __init__(self, output, root=None, interpretable=None, **kwargs):
        if root is None:
            root = output
        super().__init__(root, batch_dim=output.dims[0], **kwargs)
        self.output = output
        self.interpretable = interpretable

    def sample(
        self,
        inputs,
        n_samples=1,
        n_batch=100,
        condition=None,
        distributions=None,
        samples=None,
    ):
        self.dag = InferenceDAG().backward(
            self.output, _visited=set(inputs.copy())
        )  # don't go further than inputs

        if distributions is None:
            distributions = {}
        if condition is None:
            condition = {}

        # reset
        for component in self.dag:
            component.reset_local()

        # gather replacement values for the input variable
        values = {}

        # make sure all inputs are in the dag
        assert all(input_ in self.dag for input_ in inputs), [
            input_ in self.dag for input_ in inputs
        ]
        assert self.output in self.dag

        # first get the samples for each variable individually
        values_individual = {}

        if isinstance(n_batch, int):
            n_batch = [n_batch] * len(inputs)

        input_values = []
        for n_batch, input in zip(n_batch, inputs):
            assert (
                self.batch_dim == input.dims[0]
            ), ""  # Batch dim not in first dimension not yet supported

            if input in distributions:
                distribution = values_individual[input]
            else:
                distribution = input.distribution
            values_individual[input] = (
                distribution.spread(n_batch)
                .index_select(1, torch.tensor(0, device=input.device))
                .squeeze(1)
            )  # take first at batch dim

            input_values.append(torch.arange(values_individual[input].shape[0]))

        # now calculate the product
        prod = torch.cartesian_prod(*input_values)
        if (
            prod.ndim == 1
        ):  # cartesian product doesn't return a matrix if there's only one tensor
            prod = prod[:, None]

        n = prod.shape[0]
        new_dim = new_dim = Dim(n, name=self.batch_dim)

        # create conditional dimensions
        conditional_dimensions = {self.batch_dim: new_dim}

        # create the values for each input variable
        values = {}
        for i, (input, value_individual) in enumerate(values_individual.items()):
            definition = input.value_definition
            definition[0] = new_dim
            values[input] = definition.to_xr(
                value_individual[prod[:, i].to(value_individual.device)]
            )

        super().sample(
            n_samples=n_samples,
            values=values,
            conditional_dimensions=conditional_dimensions,
            condition=condition,
            samples=samples,
        )

    def _sample(self, dag, values, conditional_dimensions, condition, **kwargs):
        sample = {}
        for variable in dag:
            try:
                if variable in values:
                    variable.run_conditional(value=values[variable])
                else:
                    variable.run_conditional(
                        conditional_dimensions=conditional_dimensions,
                        condition=condition[variable]
                        if variable in condition
                        else None,
                    )
                self._sample_variable(variable, sample, probabilities=False, **kwargs)
            except BaseException as error:
                raise ValueError(variable.breadcrumb) from error
        return sample

    def _get_default_interpretable(self, interpretable=None, output=None):
        if interpretable is None:
            interpretable = self.interpretable

        if interpretable is not None:
            return interpretable

        if output is None:
            output = self.output

        if is_observation(output):
            return output.p.components[list(output.p.promises.keys())[0]]
        else:
            return output.components[list(output.promises.keys())[0]]



[docs]class Causal(Conditional):
    """
    A causal posterior
    """

    default_perturbed = None
    empirical = None

    def __init__(
        self,
        input,  # pylint: disable=redefined-builtin
        output,
        observed: Observed = None,
        interpretable=None,
        retain_samples=None,
        input_dependent_dims=None,
        output_sum_dims=None,
    ):
        if observed is not None:
            assert input in observed.samples
        else:
            observed = Posterior(input)

        self.input = input
        self.observed = observed

        # input sum dims
        # by default use non

        # convert dims to dim names
        if input_dependent_dims is None:
            input_dependent_dims = tuple()
        input_dependent_dims = [
            dim.name if is_dim(dim) else dim for dim in input_dependent_dims
        ]

        self.input_dependent_dims = input_dependent_dims

        # output sum dims
        # the first dimension is typically the batch dimension
        # by default, we sum over this dimension
        if output_sum_dims is None:
            output_sum_dims = output.value_definition[:1]

        # convert dim names to dims
        output_sum_dims = [dim.name if is_dim(dim) else dim for dim in output_sum_dims]
        self.output_sum_dims = output_sum_dims

        self.bootstrapped = None
        self.random = None

        if retain_samples is None:
            retain_samples = set()
        retain_samples.update(
            {
                input,
                output,
                self._get_default_interpretable(
                    interpretable=interpretable, output=output
                ),
            }
        )

        super().__init__(
            output, interpretable=interpretable, retain_samples=retain_samples
        )

    @property
    def mean_input_observed(self):
        assert self.observed is not None
        assert self.observed.samples is not None
        return self.input.support.mean(self.observed.samples[self.input], "sample")

    @property
    def mean_input(self):
        return self.samples[self.input].mean("sample")
        # return self.input.mean(self.samples[self.input])

    @property
    def std_input(self):
        return self.input.std(self.samples[self.input])

    def sample(
        self,
        n_samples: int = 1,
        n_batch: int = 100,
        distribution=None,
        condition=None,
        samples=None,
    ):
        distributions = {}
        if distribution is not None:
            distributions[self.input] = distribution
        super().sample(
            inputs=[self.input],
            n_samples=n_samples,
            n_batch=n_batch,
            distributions=distributions,
            condition=condition,
            samples=samples,
        )

    def sample_empirical(self, sample_size=200, n_bins=10, **kwargs):
        # uniformily get batches using binning
        x = self.mean_input_observed

        bins = np.linspace(x.min(), x.max(), n_bins)
        ixs = np.digitize(x, bins)

        batch_dim_index = self.output[self.batch_dim].index

        samples = []
        for bin_id in range(n_bins + 1):
            ix = ixs == bin_id
            samples.append(
                np.random.choice(
                    batch_dim_index[ix], min(sample_size, ix.sum()), replace=False
                )
            )

        coords = np.hstack(samples)

        # create and sample empirical using coords
        self.empirical = Empirical(self.output, retain_samples=self.retain_samples)
        self.empirical.sample(coords, **kwargs)

    def sample_bootstrap(
        self,
        n_samples: int = 10,
        subsample_size: int = 1000,
        subsample_n=None,
        **kwargs,
    ):
        self.bootstrapped = Bootstrapped(
            self.output, self.input, self.input_dependent_dims
        )
        self.bootstrapped.sample(
            n_samples,
            subsample_n=subsample_n,
            subsample_size=subsample_size,
            to_sum=self.output_sum_dims,
            **kwargs,
        )
        self.default_perturbed = self.bootstrapped

    def sample_random(
        self,
        n_samples: int = 10,
        subsample_size: int = 1000,
        subsample_n=None,
        **kwargs,
    ):
        self.random = Random(self.output, self.input, self.input_dependent_dims)
        self.random.sample(
            n_samples,
            subsample_size=subsample_size,
            subsample_n=subsample_n,
            to_sum=self.output_sum_dims,
            **kwargs,
        )
        self.default_perturbed = self.random

    def sample_conditional(
        self,
        n_samples: int = 10,
        subsample_size: int = 1000,
        batch_subset=None,
        **kwargs,
    ):
        self.conditional = Conditionalized(
            self.output, self.input, self.input_dependent_dims
        )
        self.conditional.sample(
            n_samples,
            subsample_size=subsample_size,
            to_sum=self.output_sum_dims,
            **kwargs,
        )
        likelihood_ratio = self.conditional.likelihood_ratio
        if batch_subset is not None:
            likelihood_ratio = likelihood_ratio.sel({self.batch_dim: batch_subset})
        self.default_perturbed = self.conditional

    @property
    def likelihood_ratio(self):
        assert self.default_perturbed is not None
        return self.default_perturbed.likelihood_ratio()

class XVectorCausal(Causal):
    @property
    def scores(self):
        scores = self.get_scores()
        if scores.index.nlevels == 1:
            scores.index = scores.index.get_level_values(0)
        return scores

    def get_scores(self):
        interpretable = self._get_default_interpretable()
        feature_dim = interpretable.value_definition[-1]

        design = crossing(
            pd.DataFrame({feature_dim.name: list(feature_dim.coords)}),
            *[
                pd.DataFrame({dim.name + "_input": list(dim.coords)})
                for dim in self.input_dependent_dims
            ],
        )

        scores = pd.DataFrame(index=pd.MultiIndex.from_frame(design))
        scores["label"] = feature_dim.label_features(
            scores.index.get_level_values(feature_dim.name)
        ).tolist()

        if self.samples is not None:
            y = (
                self.samples[interpretable]
                .mean("sample")
                .stack(
                    stacked_dim=[
                        dim.name
                        for dim in interpretable
                        if dim.name not in self.output_sum_dims
                    ]
                )
                .to_pandas()
            )

            scores["fc"] = y.max() / y.min()
            scores["ac"] = y.max() - y.min()

        if self.default_perturbed is not None:
            likelihood_ratio = self.likelihood_ratio  # .to_pandas()
            likelihood_ratio = likelihood_ratio.stack(stacked_dim=[...]).to_pandas()
            scores["lr"] = likelihood_ratio
            scores = scores.sort_values("lr", ascending=False)

        return scores
